/* ── FB2 Viewer ──────────────────────────────── */ async function loadFb2() { document.getElementById('fit-divider').style.display = 'none'; document.getElementById('btn-fit-mode').style.display = 'none'; epubViewer.style.display = 'none'; pdfViewer.style.display = 'none'; textViewer.style.display = 'block'; textViewer.style.cssText = ` display: block; flex: 1; overflow-y: auto; padding: 32px; background: var(--reader-bg-dark); color: var(--reader-text-dark); font-family: var(--font-body); font-size: 1rem; line-height: 1.9; max-width: 760px; margin: 0 auto; width: 100%; `; fontSizeBar.style.display = 'flex'; const loadingText = document.querySelector('#loading .loading-text'); loadingText.textContent = 'Loading FB2 book…'; loading.classList.remove('hidden'); try { const response = await fetch(FILE_URL); if (!response.ok) throw new Error(`HTTP ${response.status}`); const xml = await response.text(); const parser = new DOMParser(); const doc = parser.parseFromString(xml, 'text/xml'); if (doc.querySelector('parsererror')) { throw new Error('Invalid or malformed FB2 file.'); } const html = fb2ToHtml(doc); textViewer.innerHTML = html; loading.classList.add('hidden'); progressWrap.classList.remove('visible'); prevBtn.style.display = 'none'; nextBtn.style.display = 'none'; // Build TOC from chapters const chapters = doc.querySelectorAll('section > title'); tocList.innerHTML = ''; if (chapters.length === 0) { tocList.innerHTML = '
  • No chapters
    • '; } else { chapters.forEach((ch, i) => { const label = ch.textContent.trim() || `Chapter ${i + 1}`; const li = document.createElement('li'); li.className = 'toc-item'; const a = document.createElement('a'); a.href = '#'; a.textContent = label; const target = textViewer.querySelectorAll('.fb2-section')[i]; a.addEventListener('click', e => { e.preventDefault(); if (target) target.scrollIntoView({ behavior: 'smooth' }); }); li.appendChild(a); tocList.appendChild(li); }); } // Font size and theme for FB2 document.getElementById('font-dec').onclick = () => { if (currentFontSize > 60) { currentFontSize -= 10; fontSizeVal.textContent = currentFontSize + '%'; textViewer.style.fontSize = (currentFontSize / 100) + 'rem'; } }; document.getElementById('font-inc').onclick = () => { if (currentFontSize < 200) { currentFontSize += 10; fontSizeVal.textContent = currentFontSize + '%'; textViewer.style.fontSize = (currentFontSize / 100) + 'rem'; } }; } catch (err) { showError('Could not render FB2: ' + err.message); } } function fb2ToHtml(doc) { let out = ''; // Book title const bookTitle = doc.querySelector('book-title'); if (bookTitle) out += `

    ${esc(bookTitle.textContent)}

    `; // Author const fname = doc.querySelector('first-name'); const lname = doc.querySelector('last-name'); if (fname || lname) { out += `

    ${esc((fname?.textContent||'') + ' ' + (lname?.textContent||''))}

    `; } // Sections doc.querySelectorAll('body > section').forEach(section => { out += '
    '; const title = section.querySelector(':scope > title'); if (title) out += `

    ${esc(title.textContent.trim())}

    `; section.querySelectorAll('p').forEach(p => { out += `

    ${esc(p.textContent)}

    `; }); out += '
    '; }); if (!out) { // Fallback: just dump all text paragraphs doc.querySelectorAll('p').forEach(p => { out += `

    ${esc(p.textContent)}

    `; }); } return out; }
