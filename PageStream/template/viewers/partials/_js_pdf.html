/* ── PDF ──────────────────────────────────────── */ let pdfDoc = null; let pdfPageNum = 1; let pdfRendering = false; let pdfPendingPageNum = null; async function loadPdf() { epubViewer.style.display = 'none'; fontSizeBar.style.display = 'none'; pdfViewer.style.display = 'flex'; pdfViewer.style.justifyContent = 'center'; pdfViewer.style.alignItems = 'center'; document.getElementById('fit-divider').style.display = 'block'; document.getElementById('btn-fit-mode').style.display = 'flex'; try { const loadingText = document.querySelector('#loading .loading-text'); loadingText.textContent = 'Loading PDF…'; loading.classList.remove('hidden'); const pdfjsLib = await import('https://cdn.jsdelivr.net/npm/pdfjs-dist@4.4.168/build/pdf.min.mjs'); pdfjsLib.GlobalWorkerOptions.workerSrc = 'https://cdn.jsdelivr.net/npm/pdfjs-dist@4.4.168/build/pdf.worker.min.mjs'; let lastUpdate = Date.now(); const buffer = await fetchWithProgress(FILE_URL, (loaded, total) => { const now = Date.now(); if (now - lastUpdate > 100) { if (total) { const pct = Math.round((loaded / total) * 100); loadingText.textContent = `Downloading PDF… ${pct}%`; progressBar.style.width = pct + '%'; progressText.textContent = pct + '%'; } else { const mb = (loaded / (1024 * 1024)).toFixed(1); loadingText.textContent = `Downloading PDF… ${mb} MB`; } lastUpdate = now; } }); const loadingTask = pdfjsLib.getDocument({ data: buffer }); pdfDoc = await loadingTask.promise; loading.classList.add('hidden'); progressWrap.style.display = 'flex'; pdfPageNum = 1; renderPdfPage(pdfPageNum); prevBtn.onclick = () => { if (pdfPageNum <= 1) return; pdfPageNum--; renderPdfPage(pdfPageNum); }; nextBtn.onclick = () => { if (pdfPageNum >= pdfDoc.numPages) return; pdfPageNum++; renderPdfPage(pdfPageNum); }; } catch (err) { showError('Could not render PDF: ' + (err && err.message ? err.message : 'Unknown error') + '. Try downloading the file.'); } } async function renderPdfPage(num) { if (pdfRendering) { pdfPendingPageNum = num; return; } pdfRendering = true; pdfViewer.innerHTML = ''; if (pdfFitMode === 'page') { pdfViewer.style.overflowY = 'hidden'; pdfViewer.style.justifyContent = 'center'; } else { pdfViewer.style.overflowY = 'auto'; pdfViewer.style.justifyContent = 'flex-start'; } try { const page = await pdfDoc.getPage(num); const wrapperWidth = pdfViewer.clientWidth - 32; const wrapperHeight = pdfViewer.clientHeight - 32; const unscaledViewport = page.getViewport({ scale: 1 }); let scale; if (pdfFitMode === 'page') { const scaleX = wrapperWidth / unscaledViewport.width; const scaleY = wrapperHeight / unscaledViewport.height; scale = Math.min(scaleX, scaleY); } else { scale = wrapperWidth / unscaledViewport.width; } const dpr = Math.min(2, window.devicePixelRatio || 1); const viewport = page.getViewport({ scale: scale * dpr }); const canvas = document.createElement('canvas'); canvas.width = viewport.width; canvas.height = viewport.height; canvas.style.width = '100%'; canvas.style.height = 'auto'; const wrap = document.createElement('div'); wrap.className = 'pdf-page-wrap'; wrap.style.maxWidth = (viewport.width / dpr) + 'px'; wrap.appendChild(canvas); pdfViewer.appendChild(wrap); const renderContext = { canvasContext: canvas.getContext('2d'), viewport: viewport }; await page.render(renderContext).promise; pdfRendering = false; progressBar.style.width = Math.round((num / pdfDoc.numPages) * 100) + '%'; progressText.textContent = `${num} / ${pdfDoc.numPages}`; if (pdfPendingPageNum !== null) { const nextNum = pdfPendingPageNum; pdfPendingPageNum = null; renderPdfPage(nextNum); } } catch (err) { pdfRendering = false; console.error('Error rendering PDF page:', err); } }
